SELECT kursy.nazwa, COUNT(uczestnicy.id_uczestnika) AS "Zapisanych"
FROM kursy
JOIN kursy_uczestnicy ON kursy.kod = kursy_uczestnicy.kod_kursu
JOIN uczestnicy ON uczestnicy.id_uczestnika = kursy_uczestnicy.id_uczestnika
GROUP BY kursy.nazwa;
