-- zapytanie 1
SELECT tytul,plik
FROM zdjecia
WHERE polubienia >= 100;

-- zapytanie 2
SELECT zdjecia.plik,zdjecia.tytul, autorzy.imie, autorzy.nazwisko
FROM zdjecia
JOIN autorzy ON zdjecia.autorzy_id = autorzy.id
ORDER BY autorzy.nazwisko;

-- zapytanie 3
SELECT autorzy.imie, COUNT(zdjecia.id)
FROM zdjecia
JOIN autorzy ON zdjecia.autorzy_id = autorzy.id;
