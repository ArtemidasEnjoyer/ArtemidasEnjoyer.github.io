SELECT uslugi.nazwa, uslugi.cena
FROM uslugi
WHERE uslugi.cena >= 50;

-- zapytanie 2
SELECT AVG(uslugi.cena) AS "Srednia_cena", COUNT(*) AS "Ilosc_uslug"
FROM uslugi
WHERE uslugi.rodzaj = 1;
